class AttachmentMilter(ppymilterbase.PpyMilter):
def __init__(self):
	logging.info("Attachmentmilter started")
	ppymilterbase.PpyMilter.__init__(self)
	self.__mutations = []
	self.rcpt_domain = []

def OnMailFrom(self, cmd, mail_from, esmtp_info):
	self.mailFrom = mail_from
	self.__mutations = []
	return self.Continue()

def OnRcptTo(self, cmd, rcpt_to, esmtp_info):
	self.rcpt_domain.append(rcpt_to.split("@")[1])
	logging.info("rcpt_to added: %s" % rcpt_to)
	self.__mutations = []
	return self.Continue()

def OnBody(self, cmd, data):
	self.__mutations = []
	logging.info("Received email body. From: %s, To: %s", self.mailFrom, repr(self.rcpt_domain))
	if "Content-Type" in data:
		plain_html_only = False
		for line in data.splitlines(): # Check if content type is only plain, html or calendar, if so no need to block (not a real attachment, scanned by body_checks for credit card).
			if line.startswith("Content-Type: "):
				if "Content-Type: multipart/alternative;" in line or "Content-Type: text/plain;" in line or "Content-Type: text/html;" in line or "Content-Type: text/calendar;" in line:
					logging.info("Found content-type, pass due to text only. From: %s, To: %s", self.mailFrom, repr(self.rcpt_domain))
					plain_html_only = True
				else:
					logging.info("Found content-type, is real attachment. From: %s, To: %s", self.mailFrom, repr(self.rcpt_domain))
					plain_html_only = False
					break
		if plain_html_only:
			logging.info("No attachment - only html, plain or calendar. From: %s, To: %s", self.mailFrom, repr(self.rcpt_domain))
			return self.Continue()
		reject = True
		logging.info("Has attachment. From: %s, To: %s", self.mailFrom, repr(self.rcpt_domain))
		if self.mailFrom in whitelist: # Check for a specific sender
			reject = False
			for domain in self.rcpt_domain:
				if not domain in whitelist[self.mailFrom]:
					logging.info("Rejected due to: %s. From: %s, To: %s", domain, self.mailFrom, repr(self.rcpt_domain))
					reject = True
					break
#            elif not self.mailFrom.split("@")[1] in whitelist: # Check from *ALL* whitelist
#                reject = False
#                for domain in self.rcpt_domain:
#                    if not domain in whitelist["*"]: # Check if recipient domain is included in *ALL* whitelist allowing to receive emails from any domain
#                        logging.info("Rejected due to: %s. From: %s, To: %s", domain, self.mailFrom, repr(self.rcpt_domain))
#                        reject = True
#                        break
		else:
			reject = False
			for domain in self.rcpt_domain:
				if not domain in whitelist[self.mailFrom.split("@")[1]]:
					logging.info("Rejected due to: %s. From: %s, To: %s", domain, self.mailFrom, repr(self.rcpt_domain))
					reject = True
					break
		if reject:
			return self.Reject()
		else:
			logging.info("Attachment allowed. From: %s, To: %s", self.mailFrom, repr(self.rcpt_domain))
			return self.Continue()
	else:
		logging.info("No attachment. From: %s, To: %s", self.mailFrom, repr(self.rcpt_domain))
		return self.Continue()
